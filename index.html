Ejercicio 5: Suma y Promedio

//Ejercicio 5
//Se solicita al usuario ingresar la cantidad de números a promediar
let n = parseInt(prompt("Ingrese la cantidad de números que desea sumar y promediar"));
let numero;
let suma = 0;
                            
for (i = 0; i < n; i++) {
    numero = parseFloat(prompt(`Ingrese el número ${i}`));
    suma = suma + numero;
    }
                            
let promedio = suma / n;
console.log(`La suma de los números ingresados es de ${suma} y el promedio de los números es de ${promedio}.`)
